[docs]def youtube_dl_download_urls(urls: List[str]) -> None:
    ydl_opts = {
        'format': 'bestaudio/best',
        # 'logger': MyLogger(),
        'nooverwrites': True,
        'ignoreerrors': True,
    }
    if ConfigYoutubeDl.use_archive:
        ydl_opts['download_archive'] = ConfigYoutubeDl.archive_file
    with youtube_dl.YoutubeDL(ydl_opts) as ydl:
        ydl.download(urls)
